yarn workspaces turbo repo

1. clone
2. yarn

# run strapi in develop mode

1. cd apps/strapi-cms
2. cp .env.example .env
3. yarn run develop

# run scraper

1. cd apps/playwright-scraper
2. cp .env.example .env
3. yarn run scrape
